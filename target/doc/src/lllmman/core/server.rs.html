8pub struct ServerController {
9    process: Option<Child>,
10    external_pid: Option<u32>, // PID of externally running server
11    status: Arc<Mutex<ServerStatus>>,
12    provider_id: &'static str,
13}
14
15impl ServerController {
16    pub fn new(provider_id: &'static str) -> Self {
17        let mut controller = Self {
18            process: None,
19            external_pid: None,
20            status: Arc::new(Mutex::new(ServerStatus::Stopped)),
21            provider_id,
22        };
23
24        // Check for externally running server
25        controller.detect_external_server();
26
27        controller
28    }
29
30    fn detect_external_server(&mut self) {
31        // Check for llama-server process
32        if self.provider_id == "llama.cpp" {
33            if let Ok(output) = Command::new("pgrep").arg("llama-server").output() {
34                let output_str = String::from_utf8_lossy(&output.stdout);
35                if !output_str.trim().is_empty() {
36                    if let Ok(pid) = output_str.trim().parse::<u32>() {
37                        self.external_pid = Some(pid);
38                        *self.status.lock().unwrap() = ServerStatus::Running;
39                        log::info!("Detected externally running llama-server (PID {})", pid);
40                    }
41                }
42            }
43        }
44    }
45
46    pub fn is_external_running(&self) -> bool {
47        if let Some(pid) = self.external_pid {
48            // Check if process still exists
49            Command::new("kill")
50                .args(["-0", &pid.to_string()])
51                .output()
52                .map(|output| output.status.success())
53                .unwrap_or(false)
54        } else {
55            false
56        }
57    }

59    pub fn start(
60        &mut self,
61        provider: &Arc<dyn LlmProvider>,
62        config: &ProviderConfig,
63    ) -> Result<()> {
64        if self.is_running() {
65            return Err(ProviderError::ServerError("Server already running".into()));
66        }
67
68        provider.validate_config(config)?;
69
70        let mut cmd = provider.build_start_command(config);
71
72        cmd.stdin(Stdio::null());
73        cmd.stdout(Stdio::piped());
74        cmd.stderr(Stdio::piped());
75
76        let mut child = cmd.spawn()?;
77
78        if let Some(stdout) = child.stdout.take() {
79            let reader = BufReader::new(stdout);
80            std::thread::spawn(move || {
81                for line in reader.lines().flatten() {
82                    log::info!("server: {}", line);
83                }
84            });
85        }
86
87        if let Some(stderr) = child.stderr.take() {
88            let reader = BufReader::new(stderr);
89            std::thread::spawn(move || {
90                for line in reader.lines().flatten() {
91                    log::error!("server error: {}", line);
92                }
93            });
94        }
95
96        *self.status.lock().unwrap() = ServerStatus::Starting;
97        self.process = Some(child);
98
99        // Wait for server to start with timeout (up to 10 seconds, checking every 200ms)
100        let start_time = std::time::Instant::now();
101        let timeout = std::time::Duration::from_secs(10);
102
103        while start_time.elapsed() < timeout {
104            if self.is_running() {
105                *self.status.lock().unwrap() = ServerStatus::Running;
106                return Ok(());
107            }
108            std::thread::sleep(std::time::Duration::from_millis(200));
109        }
110
111        // Server failed to start within timeout
112        *self.status.lock().unwrap() =
113            ServerStatus::Error("Server failed to start within timeout".into());
114        Err(ProviderError::ServerError(
115            "Server failed to start within timeout".into(),
116        ))
117    }
118
119    pub fn stop(&mut self) -> Result<()> {
120        if let Some(mut child) = self.process.take() {
121            child.kill()?;
122            *self.status.lock().unwrap() = ServerStatus::Stopped;
123            Ok(())
124        } else {
125            Err(ProviderError::ServerError("No server running".into()))
126        }
127    }
128
129    pub fn is_running(&mut self) -> bool {
130        // Check internal process first
131        if let Some(ref mut child) = self.process {
132            if child.try_wait().ok().flatten().is_none() {
133                return true;
134            }
135        }
136
137        // Check external process
138        self.is_external_running()
139    }
140
141    pub fn get_status(&mut self) -> ServerStatus {
142        let stored_status = self.status.lock().unwrap().clone();
143
144        // If status says Running, verify process is actually alive
145        if matches!(stored_status, ServerStatus::Running) {
146            if !self.is_running() {
147                // Clear external PID if it's dead
148                if self.external_pid.is_some() {
149                    self.external_pid = None;
150                }
151                return ServerStatus::Error("Server crashed".into());
152            }
153        }
154
155        stored_status
156    }
157
158    pub fn refresh_external_detection(&mut self) {
159        // Re-detect external server (useful for periodic checks)
160        if self.provider_id == "llama.cpp" {
161            if let Ok(output) = Command::new("pgrep").arg("llama-server").output() {
162                let output_str = String::from_utf8_lossy(&output.stdout);
163                if !output_str.trim().is_empty() {
164                    if let Ok(pid) = output_str.trim().parse::<u32>() {
165                        // Only update if not already tracking this PID
166                        if self.external_pid != Some(pid) {
167                            self.external_pid = Some(pid);
168                            let current_status = self.status.lock().unwrap().clone();
169                            if !matches!(current_status, ServerStatus::Running) {
170                                *self.status.lock().unwrap() = ServerStatus::Running;
171                                log::info!(
172                                    "Detected externally running llama-server (PID {})",
173                                    pid
174                                );
175                            }
176                        }
177                    }
178                } else if self.external_pid.is_some() {
179                    // Process is gone
180                    self.external_pid = None;
181                    *self.status.lock().unwrap() = ServerStatus::Stopped;
182                }
183            }
184        }
185    }
