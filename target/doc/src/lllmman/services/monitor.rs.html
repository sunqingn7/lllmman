4pub fn get_system_stats() -> MonitorStats {
5    let mut sys = sysinfo::System::new();
6    sys.refresh_all();
7
8    let cpu = sys.cpus().first().map(|c| c.cpu_usage()).unwrap_or(0.0);
9    let total_ram = sys.total_memory() as u32;
10    let used_ram = sys.used_memory() as u32;
11
12    let gpu_usage = gpu_detector::get_all_gpu_usage();
13    let (vram_used, vram_total) = if !gpu_usage.is_empty() {
14        let gpus = gpu_detector::detect_gpus();
15        let total: u32 = gpus.iter().map(|g| g.total_vram_mb).sum();
16        let used: u32 = gpu_usage.iter().map(|u| u.used_vram_mb).sum();
17        (used, total)
18    } else {
19        (0, 0)
20    };
21
22    MonitorStats {
23        vram_used_mb: vram_used,
24        vram_total_mb: vram_total,
25        ram_used_mb: used_ram / (1024 * 1024),
26        ram_total_mb: total_ram / (1024 * 1024),
27        cpu_percent: cpu,
28        tokens_per_second: 0.0,
29        active_connections: 0,
30    }
31}
